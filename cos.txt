int liczba = 10; // przykładowa liczba całkowita
int binarna[32]; // tablica binarna do przechowywania wyniku
int i = 0; // indeks tablicy binarnej

// konwersja na system binarny
while(liczba > 0) {
    binarna[i] = liczba % 2;
    liczba = liczba / 2;
    i++;
}

// wyświetlenie wyniku w odwrotnej kolejności
for(int j = i - 1; j >= 0; j--) {
    printf("%d", binarna[j]);
}
